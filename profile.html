// প্রোফাইল পেজের স্ক্রিপ্টের ভেতরে এই অংশটুকু আপডেট করুন document.getElementById('saveProfileBtn').onclick = async () => { const name = document.getElementById('userNameInput').value; const file = document.getElementById('profilePicInput').files[0]; const submitBtn = document.getElementById('saveProfileBtn'); if(!name) return alert("Please enter your name!"); submitBtn.disabled = true; submitBtn.innerText = "Updating..."; try { const user = auth.currentUser; const userRef = doc(db, "users", user.uid); // শুধুমাত্র নাম আপডেট করার অবজেক্ট let updateData = { name: name }; // যদি নতুন ছবি সিলেক্ট করা হয় if (file) { const storageRef = ref(storage, `profiles/${user.uid}`); await uploadBytes(storageRef, file); const photoURL = await getDownloadURL(storageRef); updateData.photoURL = photoURL; // ছবির লিঙ্ক যোগ করা হলো } // updateDoc ব্যবহার করলে আপনার coins ফিল্ডটি নিরাপদ থাকবে await updateDoc(userRef, updateData); alert("Profile Updated Successfully!"); window.location.href = "profile.html"; } catch (err) { alert("Error: " + err.message); submitBtn.disabled = false; submitBtn.innerText = "Update Profile"; } };
